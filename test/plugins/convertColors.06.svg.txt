Do not apply currentColor to masks.

===

<svg xmlns="http://www.w3.org/2000/svg">
    <path fill="white"/>
    <mask id="mask1" fill="white" />
    <mask id="mask2">
        <path fill="white"/>
    </mask>
    <mask id="mask3">
        <g>
            <path fill="white"/>
            <path stroke="black"/>
        </g>
    </mask>
    <path fill="red"/>
</svg>

@@@

<svg xmlns="http://www.w3.org/2000/svg">
    <path fill="currentcolor"/>
    <mask id="mask1" fill="#fff"/>
    <mask id="mask2">
        <path fill="#fff"/>
    </mask>
    <mask id="mask3">
        <g>
            <path fill="#fff"/>
            <path stroke="#000"/>
        </g>
    </mask>
    <path fill="currentcolor"/>
</svg>

@@@

{ "currentColor": true }
